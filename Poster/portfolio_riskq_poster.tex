\begin{block}{Introduction}
\begin{itemize}
    \item Portfolio risk requires accurate tail estimates (VaR/CVaR).
    \item Classical Monte Carlo is robust but costly at scale.
    \item Portfolio RiskQ explores a classical-to-quantum workflow.
\end{itemize}
\end{block}

\begin{block}{Condensed Abstract}
We build an end-to-end pipeline to estimate portfolio tail risk, spanning data calibration, stress scenarios, nested Monte Carlo, discretization into $2^n$ bins, and an IQAE simulator that estimates tail probability with confidence intervals.
\end{block}

\begin{block}{Background}
\begin{itemize}
    \item Risk metrics: VaR and CVaR summarize tail behavior.
    \item Quantum amplitude estimation can reduce sampling cost.
    \item Discretization is the bridge between classical and quantum steps.
\end{itemize}
\end{block}

\begin{block}{Hypothesis}
IQAE-style estimation on discretized loss distributions can approximate tail probability with fewer effective samples while preserving interpretability.
\end{block}

\begin{block}{Objectives/Purpose}
\begin{itemize}
    \item Build a working classical risk engine and dashboard MVP.
    \item Add discretization and tail-mask generation.
    \item Deliver a small-scale IQAE simulator with resource reporting.
\end{itemize}
\end{block}

\begin{block}{Research}
\begin{itemize}

    \item Practical constraints for near-term quantum workflows.

\end{block}

\begin{block}{Methodology}
\textbf{Nested Monte Carlo (loss samples)}
\begin{align*}
    r \sim \mathcal{N}(\mu,\Sigma), \quad L = -w^\top r
\end{align*}
\textbf{Tail metrics}
\begin{align*}
    \text{VaR}_{\alpha} &= Q_{\alpha}(L) \\
    \text{CVaR}_{\alpha} &= \mathbb{E}[L \mid L \ge \text{VaR}_{\alpha}]
\end{align*}
\vspace{0.4em}
\textbf{Discretization}
\begin{itemize}
    \item Bin losses into $2^n$ intervals.
    \item Produce edges, probabilities, tail mask.
\end{itemize}
\vspace{0.4em}
\textbf{IQAE (simulator)}
\begin{align*}
    p &= \sin^2(\theta), \quad p_m = \sin^2((2m+1)\theta)
\end{align*}
\end{block}

\begin{block}{Analysis}
\begin{itemize}
    \item Validate probability normalization and tail-mask correctness.
    \item Compare IQAE estimates to true tail probability on toy data.
    \item Track resource usage (qubits, depth, 2Q gates, oracle calls).
\end{itemize}
\end{block}

\begin{block}{Models}
\begin{itemize}

\begin{block}{Results}
